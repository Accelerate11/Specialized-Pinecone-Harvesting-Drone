# 2. 在nano中输入以下代码（复制粘贴即可）
from pymavlink import mavutil
import time

# 连接飞控（Type-C方式）
master = mavutil.mavlink_connection('/dev/ttyACM0', baud=115200)

print("正在等待飞控心跳包...")

# 等待心跳包
master.wait_heartbeat()
print("✓ 已连接到飞控系统！")
print(f"系统ID: {master.target_system}, 组件ID: {master.target_component}")

# 循环读取姿态数据
print("\n开始读取姿态数据（按Ctrl+C退出）:\n")
while True:
    try:
        msg = master.recv_match(type='ATTITUDE', blocking=True, timeout=1)
        if msg:
            roll = msg.roll * 57.3   # 弧度转角度
            pitch = msg.pitch * 57.3
            yaw = msg.yaw * 57.3
            print(f"横滚: {roll:7.2f}° | 俯仰: {pitch:7.2f}° | 偏航: {yaw:7.2f}°")
    except KeyboardInterrupt:
        print("\n程序已退出")

# 3. 保存退出
# 按 Ctrl+O 回车保存
# 按 Ctrl+X 退出

# 4. 运行Python文件
python3 test_mavlink.py
    break
